class FakerBundle
{

    public int Property2 { get; set; }
    // Add more properties as needed
}

// Builder interface
interface IFakerBuilder
{
    void BuildProperty1();
    void BuildProperty2();
    // Add more methods to build other properties
    FakerBundle GetBundle();
}

// Concrete Builder
class ConcreteFakerBuilder : IFakerBuilder
{
    private FakerBundle bundle;

    public ConcreteFakerBuilder()
    {
        bundle = new FakerBundle();
    }

    public void BuildProperty1()
    {
        bundle.Property1 = "Example Value";
    }

    public void BuildProperty2()
    {
        bundle.Property2 = 42;
    }

    // Implement methods to build other properties

    public FakerBundle GetBundle()
    {
        return bundle;
    }
}

// Director class
class FakerDirector
{
    private IFakerBuilder builder;

    public FakerDirector(IFakerBuilder builder)
    {
        this.builder = builder;
    }

    public void ConstructBundle()
    {
        builder.BuildProperty1();
        builder.BuildProperty2();
        // Call other methods to build other properties
    }
}

// Client code
class Program
{
    static void Main(string[] args)
    {
        ConcreteFakerBuilder builder = new ConcreteFakerBuilder();
        FakerDirector director = new FakerDirector(builder);

        director.ConstructBundle();
        FakerBundle bundle = builder.GetBundle();

        Console.WriteLine("Bundle Properties:");
        Console.WriteLine($"Property1: {bundle.Property1}");
        Console.WriteLine($"Property2: {bundle.Property2}");
    }
}
